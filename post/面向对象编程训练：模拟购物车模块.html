购物车模块模拟-添加商品到购物车、查看商品信息

  1. 需求：
    • 让用户输入商品信息，并加入到购物车中去，且可立即查看购物车信息。
  2. 分析:
    • 需要让用户录入商品信息，创建对象封装商品信息。
    • 并把商品对象加入到购物车数组中去。
    • 查询购物车信息，就是便利购物车数组中的每个对象。
  3. 代码实现
    /**
  * 完成商品添加到购物车的功能
  *
  */
 private static void addGoods(Goods[] shopCar,Scanner sc) {
     //1.录入用户输入的购买商品的信息
     System.out.println("请你输入购买商品的编号（不重复）");
     int id = sc.nextInt();
     System.out.println("请你输入购买商品的名称：");
     String name = sc.next();
     System.out.println("请你输入购买商品的数量：");
     int buyNumber = sc.nextInt();
     System.out.println("请你输入购买商品的价格：");
     double price = sc.nextDouble();

     //2.把这些购买商品的信息封装成一个对象
     Goods g = new Goods();
     g.id = id;
     g.name = name;
     g.price = price;
     g.buyNumber = buyNumber;

     //3.把这个商品对象添加到宿主中去
     //shopCar = [a,a1,null,.....]
     for (int i = 0; i < shopCar.length; i++) {
         if (shopCar[i] == null){
             //说明此位置没有元素录入，把我们新买的商品添加到此处即可
             shopCar[i] = g;
             break;//不需要找位置，商品以及成功存入。
         }
     }
     System.out.println("你的商品" + g.name + "添加到购物车完成！");
 }
    /**
 * 查询购物车中的商品对象信息，并展示出来；

 */
private static void queryGoods(Goods[] shopCar) {
    System.out.println("===查询购物车信息===");
    System.out.println("编号\t\t名称\t\t\t价格\t\t\t购买数量");
    //shopCar = [g1,g2,g3,null,null...]
    for (int i = 0; i < shopCar.length; i++) {
        Goods g = shopCar[i];
        if (g != null){
            //展示这个商品
            System.out.println(g.id + "\t\t" + g.name + "\t\t" + g.price + "\t\t" + g.buyNumber);
        }else{
            //遍历结束
            break;
        }

    }
}
  4. 阶段总结
    （1）如何完成的添加商品功能？
    • 创建Goods类的对象代表商品对象，封装用户输入的商品信息。
    • 把商品第项存入到购物车的数组中去。
      （2）如何查看购物和信息？
    • 遍历购物车的数组，每遍历到一个商品对象输出其信息展示。

购物车模块训练-修改购买数量

  1. 需求：让用户输入id，找出对应商品修改其购买数量
  2. 分析：
    • 定义方法能够根据用户输入的id去购物车数组中查看是否存在该商品对象。
    • 存在返回该商品对象的地址，不存在返回null。
    • 判断返回的对象地址是否存在，存在修改其购买数量，不存在就继续。
  3. 代码实现：
    //方法：修改购买商品数量
private static void updataGoods(Goods[] shopCar,Scanner sc) {
    //让用户输入要修改商品的id
    while (true) {
        System.out.println("请你输入要修改的商品id");
        int id = sc.nextInt();
        Goods g = getGoodsByte(shopCar,id);
        if (g == null){
            //没有该商品
            System.out.println("对不起，没有购买该商品！");
        }else{
            //说明存在该商品对象，可以修改他
            System.out.println("请你输入：" + g.name + "新的购买数量");
            int buyNumber = sc.nextInt();
            g.buyNumber = buyNumber;
            System.out.println("修改完成！");
            queryGoods(shopCar);
        }
    }
}
  4. 总结
    （1）如何修改商品的数量？
    • 根据用户输入的商品id查询出要修改的商品对象。

购物车模块模拟-结算支付金额

  1. 需求：
    当用户输入pay命令后，需要展示全部购买的商品信息和总金额。
  2. 分析
    • 定义求和变量，遍历购物车数组中的全部商品，累加其单价*购买数量。
  3. 代码实现：
    //方法结算商品金额
    private static void patGoods(Goods[] shopCar) {
        queryGoods(shopCar);
        //定义一个求和变量累加金额
        double money = 0;
        //2.遍历购物车数组中的全部商品对象，累计单价*数量
        for (int i = 0; i < shopCar.length; i++) {
            Goods g = shopCar[i];
            if (g != null){
                money += (g.price * g.buyNumber);
            }else {
                break;
            }
        }
        System.out.println("订单总金额：" + money);
    }
  4. 总结
    （1）如何进行商品订单总额进行计算？
    • 定义求和变量，遍历购物车数组中的全部商品，累加其单价*购买数量。

package demo;

import java.util.Scanner;

public class ShopCarText {
    public static void main(String[] args) {
        //1.定义商品类，用于后期创建商品对象
        //2.定义购物车对象，使用一个数组对象标识。
        Goods[] shopCar = new Goods[100];
        //3.搭建操作架构
        while (true) {
            System.out.println("请你选择如下命令进行操作：");
            System.out.println("添加商品到购物车：add");
            System.out.println("查询商品到购物车：query");
            System.out.println("修改商品购买数量：update");
            System.out.println("结算购买商品金额：pay");
            System.out.println("退出系统（0）");
            Scanner sc = new Scanner(System.in);
            System.out.println("请你输入命令：");
            String command = sc.next();
            switch (command){
                case "add":
                    //添加商品到购物车
                    addGoods(shopCar,sc);
                    break;
                case "query":
                    //查询商品到购物车
                    queryGoods(shopCar);
                    break;
                case "update":
                    //修改商品购买数量
                    updataGoods(shopCar,sc);
                    break;
                case "pay":
                    //结算购买商品金额
                    patGoods(shopCar);
                    break;
                case "0":
                    System.exit(0);
                    break;
                default:
                    //输入错误
                    System.out.println("你输入的命令错误！");
                    break;
            }
        }
    }
//方法结算商品金额
    private static void patGoods(Goods[] shopCar) {
        queryGoods(shopCar);
        //定义一个求和变量累加金额
        double money = 0;
        //2.遍历购物车数组中的全部商品对象，累计单价*数量
        for (int i = 0; i < shopCar.length; i++) {
            Goods g = shopCar[i];
            if (g != null){
                money += (g.price * g.buyNumber);
            }else {
                break;
            }
        }
        System.out.println("订单总金额：" + money);
    }

    //方法：修改购买商品数量
    private static void updataGoods(Goods[] shopCar,Scanner sc) {
        //让用户输入要修改商品的id
        while (true) {
            System.out.println("请你输入要修改的商品id");
            int id = sc.nextInt();
            Goods g = getGoodsByte(shopCar,id);
            if (g == null){
                //没有该商品
                System.out.println("对不起，没有购买该商品！");
            }else{
                //说明存在该商品对象，可以修改他
                System.out.println("请你输入：" + g.name + "新的购买数量");
                int buyNumber = sc.nextInt();
                g.buyNumber = buyNumber;
                System.out.println("修改完成！");
                queryGoods(shopCar);
                break;
            }
        }
    }
    //判断商品id是否存在
    public static Goods getGoodsByte(Goods[] shopCar ,int id){
        //shopCar = [g1,g2,g3,null,null,...]
        for (int i = 0; i < shopCar.length; i++) {
            Goods g = shopCar[i];
            if (g != null){
                //判断这个商品对象的id是否是我们要找的
                if(g.id == id){
                    return g;
                }
            }else{
                    return null;
                }
            }
        return null;//代表找完100个商品也没有找到id一样的商品
    }

    /**
     * 查询购物车中的商品对象信息，并展示出来；

        System.out.println("编号\t\t名称\t\t\t价格\t\t\t购买数量");

        for (int i = 0; i < shopCar.length; i++) {

                break;
            }
        }
    }

    /**
     * 完成商品添加到购物车的功能
     *
     */
    private static void addGoods(Goods[] shopCar,Scanner sc) {
        //1.录入用户输入的购买商品的信息

        int id = sc.nextInt();
        System.out.println("请你输入购买商品的名称：");
        String name = sc.next();
        System.out.println("请你输入购买商品的数量：");
        int buyNumber = sc.nextInt();
        System.out.println("请你输入购买商品的价格：");
        double price = sc.nextDouble();

        //2.把这些购买商品的信息封装成一个对象
        Goods g = new Goods();
        g.id = id;
        g.name = name;
        g.price = price;
        g.buyNumber = buyNumber;

        //3.把这个商品对象添加到宿主中去
        //shopCar = [a,a1,null,.....]
        for (int i = 0; i < shopCar.length; i++) {
            if (shopCar[i] == null){
                //说明此位置没有元素录入，把我们新买的商品添加到此处即可
                shopCar[i] = g;
                break;//不需要找位置，商品以及成功存入。
            }
        }
        System.out.println("你的商品" + g.name + "添加到购物车完成！");
    }
}
